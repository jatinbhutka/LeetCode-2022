```python
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        
        digitToChar = {
                        '2':'abc',
                        '3':'def',
                        '4':'ghi',
                        '5':'jkl',
                        '6':'mno',
                        '7':'pqrs',
                        '8':'tuv',
                        '9':'wxyz'
                    }
        result = []
        
        def backTracking (i, currString):
            
            if len(currString) == len(digits):
                result.append(currString)
                return
            
            
            for ch in digitToChar[digits[i]]:
                backTracking(i+1, currString + ch)
                
        if len(digits) > 0:
            backTracking(0, '')
        
        return result
```
