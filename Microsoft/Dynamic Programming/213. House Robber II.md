```py
class Solution:
    def rob(self, nums: List[int]) -> int:

        # base case:
        if len(nums) == 1:
            nums[0]

        def helper(nums):
            rob2 = rob1 = 0
            for num in nums:
                temp = rob2
                rob2 = max(num + rob1, rob2)
                rob1 = temp
            return rob2
        
        # run house robber helper for subarray starting 2nd to last, 1st to second from last
        return max(helper(nums[1:]), helper(nums[:-1]), nums[0]  )
```
