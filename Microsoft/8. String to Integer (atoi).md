```python
class Solution:
    def myAtoi(self, s: str) -> int:
        
        def limitCheck(num):
            MAX_INT = 2**31 - 1
            MIN_INT = -(2**31)
            
            if num > MAX_INT:
                return MAX_INT
            elif num < MIN_INT:
                return MIN_INT
            return num
        
        result = 0
        
        # Check whiteSpace:
        i = 0        
        while i < len(s) and s[i] == " "  :
            i += 1
        
        s = s[i:]
        if len(s) == 0:
            return 0
        
        # +/- Symbol:
        sign = 1
        if s[0] == '-':
            sign = -1
        
        if s[0] == '+' or s[0] == '-':
            s = s[1:]
            
        for ch in s:
            if ch.isdigit():
                result = result * 10 + int(ch)
            else:
                return limitCheck(result * sign)
        
        return limitCheck(result * sign)    
```

Time: O(N)

Space: O(N)
